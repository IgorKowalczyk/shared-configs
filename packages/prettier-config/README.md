```bash

```

### üî© Usage

```js
// .prettierrc.js
module.exports = {
 ...require("@igorkowalczyk/prettier-config"),
};
```

```js
// prettier.config.js
import prettierConfig from "@igorkowalczyk/prettier-config";

export default {
  ...prettierConfig,
};
```

```json
{
  "extends": "@igorkowalczyk/prettier-config"
}
```
